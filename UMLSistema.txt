@startuml
class Entidad{
 -nombre: String
 -localizacion: Localizacion
 -establecimientos: List<Establecimiento>
 -tipoEntidad : TipoEntidad
}

enum TipoEntidad{
    LINEA_SUBTERRANEO
    LINEA_FERROCARRIL
    ORGANIZACION
}


class Municipio{
 +nombre: String
 +departamento: Departamento
 +provincia: Provincia
}

class Departamento{
 +nombre: String
 +provincia: Provincia
}

class Provincia{
 +nombre: String
}

class PrestadoraDeServicio{
 -usuario: Usuario
 -designado: Designado
 -entidades: List<Entidad>
}

class OrganismoDeControl{
 -usuario: Usuario
 -designado: Designado
 -entidades: List<Entidad>
}

class Establecimiento{
 -nombre: String
 -tipo: TipoEstablecimiento
 -localizacion: Localizacion
 -servicios: List<Servicio>
}

enum TipoEstablecimiento{
 ESTACION
 SUCURSAL
}

class Servicio{
    -descripcion: String
    -estado: EstadoServicio
}

enum EstadoServicio{
    DISPONIBLE
    NO_DISPONIBLE
    MANTENIMIENTO
}

class Comunidad{
    -nombre: String
    -listaAdministradores: Administrador
    -listaMiembros: Miembro
}

class Administrador{
 -persona: Persona

}

class Miembro{
 -persona: Persona
}

class Designado{
 -usuario: Usuario
}


class Persona{
    -usuario: Usuario
    -localizacionesDeInteres: List<Localizacion>
    -serviciosDeInteres: List<Servicio>
}


class Usuario{
    -username: String
    -password: String
}

class Localizacion{
 provincia: Provincia
 departamento: Departamento
 municipio: Municipio
}

abstract class GestorEntidades{
    -usuario: Usuario
    -designado: Designado
    -entidades: List<Entidades>
}

enum TipoGestor{
 Organismo
 Prestadora
}

class LectorCSV{
    -path: String
    -separator: Character

    +leerArchivo(): List<String>
    +abrirArchivo(): FileReader

}

class ParserDatos{
    -lector: LectorCSV

    +parsearLinea(linea: String[]): GestorEntidades
    +getDatos(): List<GestorEntidades>
}

Localizacion --> Provincia
Localizacion --> Departamento
Localizacion --> Municipio


Comunidad *--"*" Miembro
Comunidad *--"*" Administrador

Miembro --> Persona
Administrador --> Persona

Persona --> Usuario
Persona -->"*" Localizacion
Persona -->"*" Servicio


Entidad -->"1" Localizacion
Entidad -->"*" Establecimiento
Entidad --> TipoEntidad


Establecimiento --> TipoEstablecimiento
Establecimiento -->"1" Localizacion
Establecimiento -->"*" Servicio


OrganismoDeControl --> Usuario
OrganismoDeControl --> Designado
OrganismoDeControl -->"*" Entidad

PrestadoraDeServicio --> Usuario
PrestadoraDeServicio --> Designado
Designado --> Usuario

PrestadoraDeServicio -->"*" Entidad

Departamento --> Provincia
Municipio --> Departamento
Municipio --> Provincia

Servicio -->"1" EstadoServicio

GestorEntidades <|-- PrestadoraDeServicio
GestorEntidades <|-- OrganismoDeControl

ParserDatos --> LectorCSV
ParserDatos ..> GestorEntidades
ParserDatos ..> TipoGestor

@enduml
